1Program SIE431_ass1;


2{

4


5  Program to read in an unknown amount of data, and calculate the


6  sample mean, sample variance, and a 95% confidence interval on the


7  data.


8


9  Program should run under Turbo Pascal 5.0.


10}


11


12  var n : integer ;


13      x, sum, sumsq, ci, average, variance : real ;


14


15


16begin


17  n := 0 ;


18  sum := 0 ;


19  sumsq := 0 ;


20  x := 0 ;


21  while not eof do


22    begin


23      readln (x) ;


24      sum := sum + x ;


25      sumsq := sumsq + (x*x) ;


26      inc(n)


27    end ;


28  average := sum / n ;


29  variance := 1 / (n-1)*(sumsq - (sum*sum/n)) ;


30  ci := 1.645*sqrt(variance)/sqrt(n) ;


31  writeln ('The average is: ', average:5:3) ;


32  writeln ('The variance is: ', variance:5:3) ;


33  writeln ('A 95% confidence interval is: ', average:5:3, ' +- ', ci:5:3) ;


34  writeln ;


35  writeln ('Done!')


36end.
