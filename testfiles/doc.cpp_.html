30COle2ViewDoc::COle2ViewDoc()


31{


32    m_clsidCur = GUID_NULL ;


33    m_szObjectCur = "" ;


34	m_nType = CObjectData::typeUnknown ;


35    CString szSection ;


36    CString szKey ;


37    szSection.LoadString( IDS_INI_CONFIG ) ;


38


39    szKey.LoadString( IDS_INI_CLSCTX ) ;


40    m_dwClsCtx = (DWORD)theApp.GetProfileInt( szSection, szKey, CLSCTX_LOCAL_SERVER | CLSCTX_INPROC_SERVER ) ;


41


42    szKey = _T("ViewHiddenComCats");


43    m_fViewHiddenComCats = (BOOL)theApp.GetProfileInt( szSection, szKey, FALSE) ;


44}


45


46COle2ViewDoc::~COle2ViewDoc()


47{


48


49    CString szSection ;


50    CString szKey ;


51    szSection.LoadString( IDS_INI_CONFIG ) ;


52


53    szKey.LoadString( IDS_INI_CLSCTX ) ;


54    theApp.WriteProfileInt( szSection, szKey, (WORD)m_dwClsCtx ) ;


55


56    szKey = _T("ViewHiddenComCats");


57    theApp.WriteProfileInt( szSection, szKey, m_fViewHiddenComCats) ;


58}
